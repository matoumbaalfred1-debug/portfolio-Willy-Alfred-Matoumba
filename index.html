function calculateScore(startup, investor) {
  return (
    0.25 * matchSector(startup, investor) +
    0.20 * matchStage(startup, investor)  +
    0.15 * matchRegion(startup, investor) +
    0.15 * matchTicket(startup, investor) +
    0.15 * matchESG(startup, investor)    +
    0.10 * matchUrgency(startup)
  );
}
